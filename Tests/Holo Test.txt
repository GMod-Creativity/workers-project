@name Worker Tests
@outputs LMB Selected Money I E:entity Holos:array
runOnTick(1)
if(first()){
    Selected = 1
    Money = 500
}
if(owner():keyAttack1()){
    LMB = 1
}
else{
    LMB = 0
}
if($LMB & LMB){                         #if owner clicks
    if(Selected == 1 & Money >= 100){   #if selected is box model and there is sufficient funds
        Money -= 100                    #take 100 away
        I++                             #increment element
        holoCreate(I, owner():aimPos() + vec(0, 0, 12), vec(2, 2, 2))   #make a new holo
        holoModel(I, "cube")            #model is cube
        E = holoEntity(I)               #stores the holo just made in E
        Holos:pushEntity(E)             #pushes it onto Holos array
    }
}
